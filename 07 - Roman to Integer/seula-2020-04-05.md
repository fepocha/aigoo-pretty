# Solution

```js
/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function(s) {
    const numerals = {
        "I" : 1,
        "V" : 5,
        "X" : 10,
        "L" : 50,
        "C" : 100,
        "D" : 500,
        "M" : 1000,
    }
    
    return s.split('').reduce((prev, current, index, arr) => {
        return numerals[current] < numerals[arr[index + 1]] ? prev + (-1 * numerals[current]) : prev + numerals[current]
    }, 0)
};
```
